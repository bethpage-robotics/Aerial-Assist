001package team2869.bethpage.robotics.aerialassist;
002
003import edu.wpi.first.wpilibj.IterativeRobot;
004import edu.wpi.first.wpilibj.command.Command;
005import edu.wpi.first.wpilibj.command.Scheduler;
006import edu.wpi.first.wpilibj.livewindow.LiveWindow;
007import team2869.bethpage.robotics.aerialassist.commands.AutonomousShootBall;
008import team2869.bethpage.robotics.aerialassist.commands.CommandBase;
009
010/**
011 * The VM is configured to automatically run this class, and to call the
012 * functions corresponding to each mode, as described in the IterativeRobot
013 * documentation. If you change the name of this class or the package after
014 * creating this project, you must also update the manifest file in the resource
015 * directory.
016 * 
017 * @author BETHPAGE HIGH SCHOOL, 2014 TEAM #2869
018 */
019public class AerialAssist extends IterativeRobot {
020
021    Command autonomousCommand;
022
023    /**
024     * This function is run when the robot is first started up and should be
025     * used for any initialization code. The autonomous command is instantiated.
026     */
027    public void robotInit() {
028        // Initialize all subsystems
029        CommandBase.init();
030        // Instantiate the command used for the autonomous period.
031        autonomousCommand = new AutonomousShootBall();
032    }
033
034    public void autonomousInit() {
035        // Schedule the autonomous command.
036        autonomousCommand.start();
037    }
038
039    /**
040     * This function is called periodically during autonomous. This is all
041     * handled by the commands.
042     */
043    public void autonomousPeriodic() {
044        Scheduler.getInstance().run();
045    }
046    
047    /**
048     * This function initializes the teleoperated mode. The default commands
049     * for each subsystem are set to run, if not already running.
050     */
051    public void teleopInit() {
052        autonomousCommand.cancel();
053        
054        if (CommandBase.driveTrain.getCurrentCommand().getName().equals("MecanumDrive")) {
055            CommandBase.driveTrain.initDefaultCommand();
056        }
057        if (CommandBase.launcher.getCurrentCommand().getName().equals("WinderLaunch")) {
058            CommandBase.launcher.initDefaultCommand();
059        }
060    }
061
062    /**
063     * This function is called periodically during operator control. This is all
064     * handled by commands and operator interface control.
065     */
066    public void teleopPeriodic() {
067        Scheduler.getInstance().run();
068    }
069    
070    /**
071     * This function is called periodically during test mode. This is not called
072     * during the official game.
073     */
074    public void testPeriodic() {
075        LiveWindow.run();
076    }
077}
