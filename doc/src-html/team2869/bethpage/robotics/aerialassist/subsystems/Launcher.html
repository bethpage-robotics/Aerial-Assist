017/**
018 * This class extends Subsystem, and controls the launcher mechanism on the
019 * robot. It consists of two Talon instances for the two winding motors on the 
020 * physical launcher, as well as a release relay to shoot the ball. A limit 
021 * switch regulates the winding and unwinding of this subsystem. The subsystem 
022 * frequently updates to the SmartDashboard.
023 * 
024 * @author BETHPAGE HIGH SCHOOL, 2014 TEAM #2869 - HARSHIL GARG
025 */
026public class Launcher extends Subsystem {
027    
028    SpeedController winderL, winderR;
029    Relay releaseMotor;
030    AnalogChannel rangefinder;
031    DigitalInput limitSwitch;
032    Counter counter;
033    
034    private boolean rangeUnit = true;
035    private double rangeSlope = RobotMap.VOLT_DISTANCE_SLOPE;
036    private double rangeIntercept = RobotMap.VOLT_DISTANCE_INTERCEPT;
037   
038    private double windQuadratic = RobotMap.DISTANCE_CLICK_QUADRATIC;
039    private double windSlope = RobotMap.DISTANCE_CLICK_SLOPE;
040    private double windIntercept = RobotMap.DISTANCE_CLICK_INTERCEPT;
041    
042    private final double inchesToCentimeters = RobotMap.INCHES_CM_CONVERSION;
043    
044    private double maxShootDistance = RobotMap.MAX_SHOOT_DISTANCE;
045    private double minShootDistance = RobotMap.MIN_SHOOT_DISTANCE;
046   
047    private int currentClick = 0; 
048    private int targetClick = 0;
049    
050    /**
051     * Constructs a new Launcher, with two Talons for winding, a release
052     * relay, and a limit switch for winding regulation based on input from
053     * a Maxbotix Ultrasonic rangefinder.
054     */
055    public Launcher () {
056        super("Launcher");
057       
058        winderL = new Talon(RobotMap.WIND_LEFT_PORT);
059        winderR = new Talon(RobotMap.WIND_RIGHT_PORT);
060        
061        releaseMotor = new Relay(RobotMap.RELAY_PORT);
062        
063        rangefinder = new AnalogChannel(RobotMap.ULTRASONIC_RANGEFINDER);
064        
065        limitSwitch = new DigitalInput(RobotMap.LIMIT_SWITCH);
066        counter = new Counter(limitSwitch);
067        
068        counterInit(counter);
069    }
070    
071    /**
072     * Initializes the counter object.
073     * @param counter The counter reference.
074     */
075    private void counterInit(Counter counter) {
076        counter.start();
077        counter.reset();
078    }
079    
080    /**
081     * Initializes the default command that always runs on the Launcher, the
082     * WinderLaunch Command.
083     */
084    public void initDefaultCommand() {
085        setDefaultCommand(new WinderLaunch());
086    }
087    
088    public double getVoltage() {
089        return rangefinder.getVoltage();
090    }
091
092    public boolean getRangeUnit() {
093        return rangeUnit;
094    }
095
096    public void setRangeUnit(boolean rangeUnit) {
097        this.rangeUnit = rangeUnit;
098    }
099
100    public int getTargetClick() {
101        return targetClick;
102    }
103
104    public void setTargetClick(int targetClick) {
105        this.targetClick = targetClick;
106    }
107    
108    public int getCurrentClick() {
109        return currentClick;
110    }
111    
112    /**
113     * Shoots the ball out of the winding mechanism, by turning the release
114     * mechanism off.
115     */
116    public void shoot() {
117        setWindMotors(0);
118        releaseMotor.set(Relay.Value.kOn);
119    }
120    
121    /**
122     * Resets everyhting for the next shoot.
123     */
124    public void resetShoot() {
125        setWindMotors(0);
126        releaseMotor.set(Relay.Value.kOff);
127        counter.reset();
128        
129        currentClick = 0;
130        targetClick = 0;
131    }
132        
133    /**
134     * Calculates the distance in inches or centimeters based on a linear
135     * regression developed after testing the ultrasonic rangefinder.
136     * 
137     * @return Distance in inches, if rangeUnit is true or centimeters, if 
138     * rangeUnit is false.
139     */
140    public double calculateDistance() {
141        double distance = getVoltage()*rangeSlope + rangeIntercept;
142        if (rangeUnit)
143            return distance;
144        else
145            return inchesToCentimeters*distance;
146    }
147    
148    /**
149     * Calculates exact decimal number of target clicks. Used in isAbsoluteWound
150     * method to drive robot until an 'acceptable' error is attained.
151     * 
152     * @return Exact decimal number of target clicks.
153     */
154    private double calculateExactTargetClick() {
155        double distance;
156        
157        if (rangeUnit)
158            distance = calculateDistance();
159        else
160            distance = calculateDistance()/inchesToCentimeters;
161        
162        return windQuadratic*MathUtils.pow(distance, 2) + 
163               windSlope*distance + windIntercept;
164    }
165    
166    /**
167     * Calculates target clicks.
168     * 
169     * @return Floor of the exact number of target clicks, will result in 
170     * undershooting.
171     */
172    private int calculateTargetClick() {
173        return (int) calculateExactTargetClick();
174    }
175    
176    /**
177     * Decides if the exact target clicks and the currently wound click has an 
178     * acceptable error, to increase shooting percentage. The DriveToExactClick
179     * Command will move the robot forwards until the error is minimized.
180     * 
181     * @return True if there is an acceptable error, and false if there is not.
182     */
183    public boolean isAbsoluteWound () {
184        if (calculateExactTargetClick() - currentClick > 0.2)
185            return false;
186        else
187            return true;
188    }
189    
190    /**
191     * Decides if it safe to wind the mechanism. If the distance is too far,
192     * or the release relay is on currently, then winding might cause mechanical
193     * burnout or issues.
194     * 
195     * @return True if it is safe, and false if it is not.
196     */
197    private boolean isSafeToWind () {
198        double maxDistance, minDistance;
199        
200        if (rangeUnit) {
201            maxDistance = maxShootDistance;
202            minDistance = minShootDistance;
203        } else {
204            maxDistance = maxShootDistance*inchesToCentimeters;
205            minDistance = minShootDistance*inchesToCentimeters;
206        }
207        
208        if (calculateDistance() > maxDistance)
209            return false;
210        if (calculateDistance() < minDistance)
211            return false;
212        if (releaseMotor.get() == Relay.Value.kOn) 
213            return false;
214        
215        return true;
216    }
217    
218    /**
219     * Constantly winds and unwinds the system by regulating motor speeds based
220     * on the current clicks in comparison to target clicks. Called by the
221     * default Command, WinderLaunch, and other commands as well.
222     */
223    public void windLaunchSystem() {
224        isSwitchPressed(counter);
225        setTargetClick(calculateTargetClick());
226        
227        if (isSafeToWind()) {
228            if (currentClick < targetClick)
229                setWindMotors(1);
230            else if (currentClick > targetClick)
231                setWindMotors(-1);
232            else
233                setWindMotors(0);
234        }
235    }
236    
237    /**
238     * Determines the action of the drivetrain system during autonomous 
239     * game mode.
240     * 
241     * @param autonomousDistance The distance, in inches, that the robot should 
242     * be from the goal before shooting.
243     * @return A positive value to indicate forward drive, zero to indicate stop 
244     * (target range achieved) and a negative value to indicate backward drive.
245     */
246    public int autonomousDrive(double autoDistance) {
247        double autonomousDistance;
248        
249        if (rangeUnit)
250            autonomousDistance = autoDistance;
251        else
252            autonomousDistance = autoDistance*inchesToCentimeters;
253        
254        if (calculateDistance() - autonomousDistance > 3)
255            return 1;
256        else if (calculateDistance() - autonomousDistance < -3)
257            return -1;
258        else 
259            return 0;
260    }
261    
262    /**
263     * Determines which of the two goals is currently lit (at 0 seconds), and 
264     * proceeds to the other goal. Method is currently incomplete.
265     * 
266     * @return A positive value to indicate that to proceed to the right goal,
267     * and a negative value to proceed to the left goal.
268     */
269    public int autonomousDirection() {
270        return 1;
271    }
272    
273    /**
274     * Decides if the winders are moving by analyzing if they are within an 
275     * acceptable speed range.
276     * 
277     * @return True if either of the winders are moving, and false if not.
278     */
279    private boolean winderMoving () {
280        if (winderL.get() > 0.1 && winderL.get() < -0.1)
281            return true;
282        else if (winderR.get() > 0.1 && winderR.get() < -0.1)
283            return true;
284        else
285            return false;
286    }    
287    /**
288     * Decides if the mechanism is done winding. The winding motors must be set
289     * to zero, if currentClicks equals targetClicks. Safety check, since if the
290     * shooting mechanism is called, the other Commands would have successfully
291     * wound the mechanism by now.
292     * 
293     * @return True if the mechanism is ready to shoot, and false if it is not.
294     */
295    public boolean isDoneWinding() {
296        if (currentClick == targetClick && !winderMoving())
297            return true;
298        else
299            return false;
300    }
301    
302    /**
303     * Sets the winder motor values.
304     * @param value The motor speed to set both motors to.
305     */
306    private void setWindMotors(double value) {
307        winderL.set(value);
308        winderR.set(value);
309    }
310    
311    /**
312     * Measures a physical switch press and translates it to virtual meaning. If
313     * the winder motor is moving in the negative direction, the mechanism is 
314     * unwinding and so the switch press count should decrease, and if the motor
315     * is moving in the positive direction, the mechanism is winding, and thus 
316     * the switch press count should increase.
317     * 
318     * @param counter The Counter object that represents the physical limit
319     * switch virtually. A physical switch press increments the counter value
320     * twice: once on the press (by 4 - 10) and once on the release (by 1 - 3).
321     * The code only acknowledges the presses by ignoring all increases that
322     * are less than 4.
323     */
324    private void isSwitchPressed(Counter counter) {
325        if (counter.get() > 3) {
326            if (winderL.get() <= -0.1) {
327                currentClick--;
328            }
329            else if (winderL.get() >= 0.1) {
330                currentClick++;
331            }
332            else {
333                System.out.println("The switch should not have been pressed.");
334            }
335            counter.reset();
336        }
337    }
338    
339    /**
340     * Allows the printing of the relay's status to SmartDashboard by returning
341     * a String
342     * @return The state of the relay: On or Off.
343     */
344    private String relayStatus() {
345        if (releaseMotor.get() == Relay.Value.kOn) {
346            return "On";
347        }
348        else {
349            return "Off";
350        }
351    }
352    
353    /**
354     * Updates the SmartDashboard with winder motor values, relay status, the
355     * distance to wall, and the state of the launcher mechanism.
356     */
357    public void updateDashboard() {
358        SmartDashboard.putNumber("Winder left", winderL.get());
359        SmartDashboard.putNumber("Winder right", winderR.get());
360        SmartDashboard.putString("Relay", relayStatus());
361        
362        SmartDashboard.putNumber("Rangefinder voltage", getVoltage());
363        SmartDashboard.putNumber("Rangefinder distance", calculateDistance());
364        
365        SmartDashboard.putNumber("Current clicks", getCurrentClick());
366        SmartDashboard.putNumber("Exact target clicks", calculateExactTargetClick());
367        SmartDashboard.putNumber("Target clicks", calculateTargetClick());
368    }
