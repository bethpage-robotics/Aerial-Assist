001package team2869.bethpage.robotics.aerialassist;
002
003import edu.wpi.first.wpilibj.AnalogChannel;
004
005/**
006 * This Ultrasonic class stores the linear regression to transform voltages into
007 * distances.
008 *
009 * @author BETHPAGE HIGH SCHOOL, 2014 TEAM #2869 - HARSHIL GARG
010 */
011public class Ultrasonic extends AnalogChannel {
012
013    private double distanceSlope = RobotMap.VOLT_DISTANCE_SLOPE;
014    private double distanceIntercept = RobotMap.VOLT_DISTANCE_INTERCEPT;
015
016    /**
017     * Constructs an Ultrasonic rangefinder that is wired at the specific
018     * channel
019     *
020     * @param channel The analog breakout board channel that the rangefinder is
021     * connected to.
022     */
023    public Ultrasonic(int channel) {
024        super(channel);
025    }
026
027    /**
028     * Constructs an Ultrasonic rangefinder that is wired at the specific module
029     * number and channel.
030     *
031     * @param moduleNumber The module of the analog breakout board.
032     * @param channel The analog breakout board channel that the rangefinder is
033     * connected to.
034     */
035    public Ultrasonic(int moduleNumber, int channel) {
036        super(moduleNumber, channel);
037    }
038    
039    /**
040     * Finds the voltage of the Ultrasonic rangefinder. Voltage scales with
041     * increasing distance.
042     * 
043     * @return A voltage from 0V to approximately 2.5V.
044     */
045    public double getVoltage() {
046        return super.getVoltage();
047    }
048    
049    /**
050     * Calculates the distance in inches or centimeters based on a linear
051     * regression developed after testing the ultrasonic rangefinder.
052     * 
053     * @return Distance in inches.
054     */
055    public double getDistance() {
056        return getVoltage() * distanceSlope + distanceIntercept;
057    }
058}
