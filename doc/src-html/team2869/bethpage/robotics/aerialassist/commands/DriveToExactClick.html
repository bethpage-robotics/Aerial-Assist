001package team2869.bethpage.robotics.aerialassist.commands;
002
003/**
004 * Often the robot may be positioned by the operator so that the target power
005 * is between two integer click values. This Command drives the robot forward 
006 * until the lower click bound is reached, to increase shot accuracy quite
007 * tremendously.
008 * 
009 * @author BETHPAGE HIGH SCHOOL, 2014 TEAM #2869 - HARSHIL GARG
010 */
011public class DriveToExactClick extends CommandBase {
012    
013    /**
014     * Constructs a new Command.
015     */
016    public DriveToExactClick() {
017        requires(driveTrain);
018        requires(launcher);
019        
020        this.setInterruptible(false);
021    }
022
023    protected void initialize() {
024    }
025
026    /**
027     * Drives the robot forward slowly (to minimize jerk) and continues to wind
028     * the launch system if necessary, while updating the SmartDashboard.
029     */
030    protected void execute() {
031        driveTrain.mecanumDrive(0, 0.25, 0, 0);
032        launcher.windLaunchSystem();
033        launcher.updateDashboard();
034    }
035
036    /**
037     * If the shooting mechanism has wound to an absolute click value, that is 
038     * much closer to a target integer value, then this command can stop
039     * executing.
040     * 
041     * @return True if the command should terminate, and false if not.
042     */
043    protected boolean isFinished() {
044        return launcher.isAbsoluteWound();
045    }
046
047   /**
048    * Resets the driveTrain to stop moving because when the shot is made, the
049    * driveTrain should not move and be inaccessible to the operator.
050    */
051    protected void end() {
052        driveTrain.mecanumDrive(0, 0, 0, 0);
053    }
054    
055    protected void interrupted() {
056    }
057}
