001package team2869.bethpage.robotics.aerialassist.commands;
002
003import edu.wpi.first.wpilibj.command.Command;
004import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
005
006import team2869.bethpage.robotics.aerialassist.OI;
007import team2869.bethpage.robotics.aerialassist.subsystems.DriveTrain;
008import team2869.bethpage.robotics.aerialassist.subsystems.ExampleSubsystem;
009import team2869.bethpage.robotics.aerialassist.subsystems.Launcher;
010
011/**
012 * The base for all commands. All atomic commands should subclass CommandBase.
013 * CommandBase stores creates and stores each control system. To access a
014 * subsystem elsewhere in the code, use CommandBase.subsystem
015 * 
016 * @author BETHPAGE HIGH SCHOOL, 2014 TEAM #2869
017 */
018public abstract class CommandBase extends Command {
019
020    public static OI operatorInterface;
021    
022    // Create a single static instance of all subsystems
023    public static ExampleSubsystem exampleSubsystem = new ExampleSubsystem();
024    public static DriveTrain driveTrain = new DriveTrain();
025    public static Launcher launcher = new Launcher();
026    
027    /**
028     * Initialize the operator interface and show which commands are running on
029     * the various subsystems on the SmartDashboard.
030     */
031    public static void init() {
032        // Instantiate the operator interface
033        operatorInterface = new OI();
034
035        // Show command the subsystem is running on the SmartDashboard
036        SmartDashboard.putData(driveTrain);
037        SmartDashboard.putData(launcher);
038        SmartDashboard.putData(exampleSubsystem);
039    }
040    
041    /**
042     * Constructor for the CommandBase class
043     * @param name The name of the class.
044     */
045    public CommandBase(String name) {
046        super(name);
047    }
048    
049    /**
050     * Default constructor for the CommandBase class.
051     */
052    public CommandBase() {
053        super();
054    }
055}
