001package team2869.bethpage.robotics.aerialassist.commands;
002
003/**
004 * Default Command that constantly runs on the launcher subsystem until another
005 * Command replaces it. Allows for constant winding functionality.
006 * 
007 * @author BETHPAGE HIGH SCHOOL, 2014 TEAM #2869 - HARSHIL GARG
008 */
009public class WinderLaunch extends CommandBase {
010    
011    public WinderLaunch() {
012        requires(launcher);
013    }
014
015    protected void initialize() {
016    }
017    
018    /**
019     * Winds the launcher while continually updating the SmartDashboard.
020     */
021    protected void execute() {
022        launcher.windLaunchSystem();
023        launcher.updateDashboard();
024    }
025
026    /**
027     * Always returns false, since this is the default Command.
028     * 
029     * @return False
030     */
031    protected boolean isFinished() {
032        return false;
033    }
034
035    protected void end() {
036    }
037
038    protected void interrupted() {
039    }
040}
