001package team2869.bethpage.robotics.aerialassist.commands;
002
003/**
004 * Command that inverts the orientation of the robot, and is quite a deal more
005 * intuitive than spinning the robot around.
006 * 
007 * @author BETHPAGE HIGH SCHOOL, 2014 TEAM #2869 - HARSHIL GARG
008 */
009public class InvertedDrive extends CommandBase {
010    
011    /**
012     * Constructs a new Command.
013     */
014    public InvertedDrive() {
015        requires(driveTrain);
016        this.setInterruptible(false);
017    }
018
019    /**
020     * Changes the orientation of the motor input, so that all signals are
021     * flipped, artificially creating an inverted drive state.
022     */
023    protected void initialize() {
024        driveTrain.setOrientationLeftWheels(true);
025        driveTrain.setOrientationRightWheels(false);
026    }
027
028    /**
029     * Drives the robot with regular mecanum style code, while accounting for
030     * slight joystick changes. The SmartDashboard is also continually updated.
031     */
032    protected void execute() {
033        double cartesianX = operatorInterface.getCartesianX(),
034               cartesianY = operatorInterface.getCartesianY(),
035               rotation = operatorInterface.getRotation();
036        double x, y, r;
037        if (Math.abs(cartesianX) < 0.1) {
038            x = 0;
039        } else {
040            x = cartesianX;
041        }
042        if (Math.abs(cartesianY) < 0.1) {
043            y = 0;
044        } else {
045            y = cartesianY;
046        }
047        if (Math.abs(rotation) < 0.1) {
048            r = 0;
049        } else {
050            r = rotation;
051        }
052
053        driveTrain.mecanumDrive(x, y, r, 0);
054        driveTrain.updateDashboard(cartesianX, cartesianY, rotation);
055    }
056
057    /**
058     * If the toggle button that controls the execution of this Command is
059     * toggled off, then this Command will be terminated.
060     * 
061     * @return True if the command should terminate, and false if not.
062     */
063    protected boolean isFinished() {
064        return false;
065    }
066
067    protected void end() {
068    }
069
070    protected void interrupted() {
071    }
072}
