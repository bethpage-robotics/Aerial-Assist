003/**
004 * The RobotMap is a mapping from the ports sensors and actuators are wired into
005 * to a variable name. This provides flexibility changing wiring and also makes 
006 * checking the wiring easier.
007 */
008public class RobotMap {
009    // Joysticks: This is the mapping of the joystick ports in the computer.
010    public static final int MECANUM_STICK_PORT = 1,
011                            ROTATE_STICK_PORT = 2;
012    
013    // Motors: This is the mapping of motors ports marked PWM on the DIO board
014    public static final int FRONT_LEFT_PORT = 2,
015                            FRONT_RIGHT_PORT = 1,
016                            REAR_LEFT_PORT = 4,
017                            REAR_RIGHT_PORT = 3;
018    
019    public static final int WIND_LEFT_PORT = 5,
020                            WIND_RIGHT_PORT = 6;
021    
022    public static final int RELAY_PORT = 1;
023    
024    // Sensors: This is the mapping of the analog ports on the analog module
025    // to the relevant sensors.
026    public static final int ULTRASONIC_RANGEFINDER = 1,
027                            LIMIT_SWITCH = 1;
028    
029    // Buttons: This is the mapping of the buttons on the main joystick.
030    public static final int SPIN_BUTTON = 2,
031                            STOP_BUTTON = 3,
032                            HALF_BUTTON = 4;
033    
034    public static final int TURN_RIGHT_BUTTON = 5,

037    
038    public static final int INVERTED_DRIVE_BUTTON = 8;
039    
040    public static final int SHOOTBALL_BUTTON = 9;
041    
042    // Constants: This is the mapping of all required constants.
043    public static final double VOLT_DISTANCE_SLOPE = .0096560282;
044    public static final double VOLT_DISTANCE_INTERCEPT = -.0015728398;
